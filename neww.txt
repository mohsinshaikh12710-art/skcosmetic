<script>
const PRODUCTS = [
  {id:1,title:"Classic Men T-Shirt",price:499,img:"https://via.placeholder.com/400x400?text=T-shirt"},
  {id:2,title:"Leather Wallet",price:799,img:"https://via.placeholder.com/400x400?text=Wallet"},
  {id:3,title:"Stylish Sunglasses",price:899,img:"https://via.placeholder.com/400x400?text=Sunglasses"},
  {id:4,title:"Sport Shoes",price:1999,img:"https://via.placeholder.com/400x400?text=Shoes"},
  {id:5,title:"Backpack",price:1299,img:"https://via.placeholder.com/400x400?text=Backpack"},
  {id:6,title:"Wrist Watch",price:1599,img:"https://via.placeholder.com/400x400?text=Watch"}
];

let cart = {};

function format(n){ return n.toFixed(2) }

function renderProducts(){
  const grid = document.getElementById('productGrid');
  grid.innerHTML = '';
  PRODUCTS.forEach(p=>{
    const card = document.createElement('div'); card.className='card';
    card.innerHTML = `
      <img src="${p.img}" alt="${p.title}">
      <h3>${p.title}</h3>
      <div class="small">Some short description here.</div>
      <div style="display:flex;justify-content:space-between;align-items:center;margin-top:8px">
        <div class="price">₹${p.price}</div>
        <button class="btn" onclick="addToCart(${p.id})">Add to cart</button>
      </div>
    `;
    grid.appendChild(card);
  });
  document.getElementById('count').innerText = PRODUCTS.length;
}

function addToCart(id){
  cart[id] = cart[id] ? cart[id]+1 : 1;
  renderCart();
  showCart();
}

function renderCart(){
  const list = document.getElementById('cartList');
  list.innerHTML = '';
  let total = 0;
  for(const id in cart){
    const p = PRODUCTS.find(x=>x.id==id);
    const qty = cart[id];
    const lineTotal = p.price * qty;
    total += lineTotal;
    const div = document.createElement('div'); div.className='cart-item';
    div.innerHTML = `
      <img src="${p.img}" style="width:48px;height:48px;border-radius:6px;object-fit:cover">
      <div style="flex:1">
        <div style="font-weight:600">${p.title}</div>
        <div class="small">₹${p.price} x ${qty} = ₹${lineTotal}</div>
      </div>
      <div style="display:flex;flex-direction:column;gap:6px">
        <button style="padding:6px;border-radius:6px;border:0;background:#eee;cursor:pointer" onclick="changeQty(${id},1)">+</button>
        <button style="padding:6px;border-radius:6px;border:0;background:#eee;cursor:pointer" onclick="changeQty(${id},-1)">-</button>
      </div>
    `;
    list.appendChild(div);
  }
  document.getElementById('cartTotal').innerText = format(total);
  document.getElementById('cart').style.display = Object.keys(cart).length ? 'block' : 'none';
}

function changeQty(id,delta){
  cart[id] = (cart[id]||0) + delta;
  if(cart[id] <= 0) delete cart[id];
  renderCart();
}

function showCart(){ document.getElementById('cart').style.display = 'block' }

function checkout(){
  if(!Object.keys(cart).length){ alert('Cart is empty'); return }
  alert('Demo checkout — integrate payment gateway to process orders.');
}

renderProducts();
</script>
